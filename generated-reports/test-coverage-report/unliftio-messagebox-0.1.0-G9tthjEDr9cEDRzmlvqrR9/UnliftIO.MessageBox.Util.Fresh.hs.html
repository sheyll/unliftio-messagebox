   25 -- | Atomically increment and get the value of the 'Counter'
   26 -- for type @a@ that must be present in the @env@.
   27 {-# INLINE fresh #-}
   28 fresh ::
   29   forall a env m.
   30   ( MonadReader env m,
   31     MonadIO m,
   32     HasCounterVar a env,
   33     Coercible a Int
   34   ) =>
   35   m a
   36 fresh =
   37   asks (getCounterVar @a) >>= incrementAndGet

   39 -- | Atomically increment and get the value of the 'Counter'
   40 -- for type @a@ that must be present in the @env@.
   41 {-# INLINE incrementAndGet #-}
   42 incrementAndGet ::
   43   forall a m.
   44   ( MonadIO m,
   45     Coercible a Int
   46   ) =>
   47   CounterVar a -> m a
   48 incrementAndGet (MkCounterVar !atomicCounter) =
   49   coerce <$> liftIO (incrCounter 1 atomicCounter)

   52 -- | Create a new 'CounterVar' starting at @0@.
   53 {-# INLINE newCounterVar #-}
   54 newCounterVar ::
   55   forall a m.
   56   MonadIO m =>
   57   m (CounterVar a)
   58 newCounterVar =
   59   MkCounterVar <$> liftIO (newCounter 0)
