    1 -- | This module contains a type class that
    2 -- describes exchangable operations on messages
    3 -- boxes.
    4 module UnliftIO.MessageBox.Class
    5   ( IsMessageBoxFactory (..),
    6     IsMessageBox (..),
    7     IsInput (..),
    8     handleMessage,
    9   )
   10 where
   11 
   12 import Data.Kind (Type)
   13 import UnliftIO.MessageBox.Util.Future (Future, awaitFuture)
   14 import UnliftIO (MonadUnliftIO, timeout)

   16 -- | Create 'IsMessageBox' instances from a parameter.
   17 -- Types that determine 'MessageBox' values.
   18 --
   19 -- For a limited message box this might be the limit of
   20 -- the message queue.
   21 class
   22   (IsMessageBox (MessageBox cfg), IsInput (Input (MessageBox cfg))) =>
   23   IsMessageBoxFactory cfg
   24   where
   25   type MessageBox cfg :: Type -> Type
   26 
   27   -- | Return a message limit.
   28   --
   29   -- NOTE: This method was added for unit tests.
   30   -- Although the method is totally valid, it
   31   -- might not be super useful in production code.
   32   -- Also note that the naming follows the rule:
   33   -- Reserve short names for entities that are
   34   -- used often.
   35   getConfiguredMessageLimit :: cfg -> Maybe Int
   36 
   37   -- | Create a new @msgBox@.
   38   -- This is required to receive a message.
   39   -- NOTE: Only one process may receive on an msgBox.
   40   newMessageBox :: MonadUnliftIO m => cfg -> m (MessageBox cfg a)
   41 
   42 -- | A type class for msgBox types.
   43 -- A common interface for receiving messages.
   44 class IsInput (Input msgBox) => IsMessageBox msgBox where
   45   -- | Type of the corresponding input
   46   type Input msgBox :: Type -> Type
   47 
   48   -- | Receive a message. Take whatever time it takes.
   49   -- Return 'Just' the value or 'Nothing' when an error
   50   -- occurred.
   51   --
   52   -- NOTE: Nothing may sporadically be returned, especially
   53   -- when there is a lot of load, so please make sure to 
   54   -- build your application in such a way, that it 
   55   -- anticipates failure.
   56   receive :: MonadUnliftIO m => msgBox a -> m (Maybe a)
   57 
   58   -- | Return a 'Future' that can be used to wait for the
   59   -- arrival of the next message.
   60   -- NOTE: Each future value represents the next slot in the queue
   61   -- so one future corresponds to exactly that message (should it arrive)
   62   -- and if that future value is dropped, that message will be lost!
   63   tryReceive :: MonadUnliftIO m => msgBox a -> m (Future a)
   64 
   65   -- | Wait for an incoming message or return Nothing.
   66   --
   67   -- The default implementation uses 'tryReceive' to get a
   68   -- 'Future' on which 'awaitFuture' inside a 'timeout' is called.
   69   --
   70   -- Instances might override this with more performant implementations
   71   -- especially non-blocking Unagi channel based implementation.
   72   --
   73   -- NOTE: Nothing may sporadically be returned, especially
   74   -- when there is a lot of load, so please make sure to 
   75   -- build your application in such a way, that it 
   76   -- anticipates failure.
   77   receiveAfter ::
   78     MonadUnliftIO m =>
   79     -- | Message box
   80     msgBox a ->
   81     -- | Time in micro seconds to wait until the
   82     -- action is invoked.
   83     Int ->
   84     m (Maybe a)
   85   receiveAfter !mbox !t =
   86     tryReceive mbox >>= timeout t . awaitFuture
   87 
   88   -- | Create a new @input@ that enqueus messages,
   89   -- which are received by the @msgBox@
   90   newInput :: MonadUnliftIO m => msgBox a -> m (Input msgBox a)
   91 
   92 -- | A type class for input types.
   93 -- A common interface for delivering messages.
   94 class IsInput input where
   95   -- | Send a message. Take whatever time it takes.
   96   -- Depending on the implementation, this might
   97   -- be a non-blocking operation.
   98   -- Return if the operation was successful.
   99   --
  100   -- NOTE: @False@ may sporadically be returned, especially
  101   -- when there is a lot of load, so please make sure to 
  102   -- build your application in such a way, that it 
  103   -- anticipates failure.
  104   deliver :: MonadUnliftIO m => input a -> a -> m Bool

  108 -- | Receive a message and apply a function to it.
  109 --  TODO exception handling: indefinitely blocked in mvar... is it really necessary to catch here?
  110 handleMessage ::
  111   (MonadUnliftIO m, IsMessageBox msgBox) =>
  112   msgBox message ->
  113   (message -> m b) ->
  114   m (Maybe b)
  115 handleMessage !msgBox !onMessage = do
  116   !maybeMessage <- receive msgBox
  117   case maybeMessage of
  118     Nothing -> pure Nothing
  119     Just !message -> do
  120       Just <$> onMessage message
