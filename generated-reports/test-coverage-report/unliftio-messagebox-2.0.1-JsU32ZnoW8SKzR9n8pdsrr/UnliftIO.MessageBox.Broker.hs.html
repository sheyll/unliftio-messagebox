    3 -- | A broker extracts a /key/ value from incoming messages
    4 --  and creates, keeps and destroys a /resource/ for each key.
    5 --
    6 -- The demultiplexed messages and their resources are passed to
    7 -- a custom 'MessageHandler'/
    8 --
    9 -- The user provides a 'Demultiplexer' is a pure function that
   10 -- returns a key for the resource associated
   11 -- to the message and potientially changes the
   12 -- message.
   13 --
   14 -- The demultiplexer may also return a value indicating that
   15 -- a new resource must be created, or that a message
   16 -- shall be ignored.
   17 --
   18 -- The broker is run in a seperate process using 'async'.
   19 -- The usual way to stop a broker is to 'cancel' it.
   20 --
   21 -- When cancelling a broker, the resource cleanup
   22 -- actions for all resources will be called with
   23 -- async exceptions masked.
   24 --
   25 -- In order to prevent the resource map filling up with
   26 -- /dead/ resources, the user of this module has to ensure
   27 -- that whenever a resource is not required anymore, a message
   28 -- will be sent to the broker, that will cause the 'MessageHandler'
   29 -- to be executed for the resource, which will in turn return,
   30 -- return 'RemoveResource'.
   31 module UnliftIO.MessageBox.Broker
   32   ( spawnBroker,
   33     BrokerConfig (..),
   34     BrokerResult (..),
   35     ResourceCreator,
   36     Demultiplexer,
   37     ResourceCleaner,
   38     MessageHandler,
   39     Demuxed (..),
   40     ResourceUpdate (..),
   41   )
   42 where

   64 -- | Spawn a broker with a new 'MessageBox',
   65 --  and return its message 'Input' channel as well as
   66 -- the 'Async' handle of the spawned process, needed to
   67 -- stop the broker process.
   68 --
   69 -- * @k@ is the /key/ for the resource associated to an incoming
   70 --   message
   71 -- * @w'@ is the type of incoming messages.
   72 -- * @w@ is the type of the demultiplexed messages.
   73 -- * @a@ specifies the resource type.
   74 -- * @m@ is the base monad
   75 spawnBroker ::
   76   forall brokerBoxArg k w' w a m.
   77   ( MonadUnliftIO m,
   78     Ord k,
   79     IsMessageBoxArg brokerBoxArg
   80   ) =>
   81   brokerBoxArg ->
   82   BrokerConfig k w' w a m ->
   83   m
   84     ( Either
   85         SomeException
   86         ( Input (MessageBox brokerBoxArg) w',
   87           Async BrokerResult
   88         )
   89     )
   90 spawnBroker brokerBoxArg config = do
   91   brokerA <- async $ do
   92     mBrokerBox <-
   93       tryAny
   94         ( do
   95             b <- newMessageBox brokerBoxArg
   96             i <- newInput b
   97             return (b, i)
   98         )
   99     case mBrokerBox of
  100       Left er -> return (Left er)
  101       Right (brokerBox, brokerInp) -> do
  102         aInner <- mask_ $
  103           asyncWithUnmask $ \unmaskInner ->
  104             brokerLoop unmaskInner brokerBox config Map.empty
  105         return (Right (brokerInp, aInner))
  106   join <$> waitCatch brokerA

  108 -- | This is just what the 'Async' returned from
  109 -- 'spawnBroker' returns, it's current purpose is to
  110 -- make code easier to read.
  111 --
  112 -- Instead of some @Async ()@ that could be anything,
  113 -- there is @Async BrokerResult@.
  114 data BrokerResult = MkBrokerResult
  115   deriving stock (Show, Eq)

  117 -- | The broker configuration, used by 'spawnBroker'.
  118 --
  119 -- * @k@ is the /key/ for the resource associated to an incoming
  120 --   message
  121 -- * @w'@ is the type of incoming messages.
  122 -- * @w@ is the type of the demultiplexed messages.
  123 -- * @a@ specifies the resource type.
  124 -- * @m@ is the base monad
  125 data BrokerConfig k w' w a m = MkBrokerConfig
  126   { demultiplexer :: !(Demultiplexer w' k w),
  127     messageDispatcher :: !(MessageHandler k w a m),
  128     resourceCreator :: !(ResourceCreator k w a m),
  129     resourceCleaner :: !(ResourceCleaner k a m)
  130   }
  131 
  132 -- | User supplied callback to extract the key and the 'Demuxed'
  133 --  from a message.
  134 --  (Sync-) Exceptions thrown from this function are caught and lead
  135 --  to dropping of the incoming message, while the broker continues.
  136 --
  137 -- * @k@ is the /key/ for the resource associated to an incoming
  138 --   message
  139 -- * @w'@ is the type of incoming messages.
  140 -- * @w@ is the type of the demultiplexed messages.
  141 type Demultiplexer w' k w = w' -> Demuxed k w
  142 
  143 -- | User supplied callback to use the 'Demuxed' message and
  144 --  the associated resource.
  145 --  (Sync-) Exceptions thrown from this function are caught and lead
  146 --  to immediate cleanup of the resource but the broker continues.
  147 --
  148 -- * Type @k@ is the /key/ for the resource associated to an incoming
  149 --   message
  150 -- * Type @w@ is the type of incoming, demultiplexed, messages.
  151 -- * Type @a@ specifies the resource type.
  152 -- * Type @m@ is the base monad
  153 type MessageHandler k w a m = k -> w -> a -> m (ResourceUpdate a)
  154 
  155 -- | This value indicates in what state a worker is in after the
  156 --  'MessageHandler' action was executed.
  157 data ResourceUpdate a
  158   = -- | The resources is still required.
  159     KeepResource
  160   | -- | The resource is still required but must be updated.
  161     UpdateResource a
  162   | -- | The resource is obsolete and can
  163     --   be removed from the broker.
  164     --   The broker will call 'ResourceCleaner' either
  165     --   on the current, or an updated resource value.
  166     RemoveResource !(Maybe a)
  167 
  168 -- | The action that the broker has to take for in incoming message.
  169 --
  170 -- * @k@ is the /key/ for the resource associated to an incoming
  171 --   message
  172 -- * @w@ is the type of the demultiplexed messages.
  173 data Demuxed k w
  174   = -- | The message is an initialization message, that requires the
  175     --   creation of a new resouce for the given key.
  176     --   When the resource is created, then /maybe/ additionally
  177     --   a message will also be dispatched.
  178     Initialize k !(Maybe w)
  179   | -- | Dispatch a message using an existing resource.
  180     -- Silently ignore if no resource for the key exists.
  181     Dispatch k w
  182 
  183 -- | User supplied callback to create and initialize a resource.
  184 --  (Sync-) Exceptions thrown from this function are caught,
  185 --  and the broker continues.
  186 --
  187 -- * @k@ is the /key/ for the resource associated to an incoming
  188 --   message
  189 -- * @w@ is the type of the demultiplexed messages.
  190 -- * @a@ specifies the resource type.
  191 -- * @m@ is the monad of the returned action.
  192 type ResourceCreator k w a m = k -> Maybe w -> m a
  193 
  194 -- | User supplied callback called _with exceptions masked_
  195 -- when the 'MessageHandler' returns 'RemoveResource'
  196 -- (Sync-) Exceptions thrown from this function are caught,
  197 -- and do not prevent the removal of the resource, also the
  198 -- broker continues.
  199 --
  200 -- * @k@ is the /key/ for the resource associated to an incoming
  201 --   message
  202 -- * @a@ specifies the resource type.
  203 -- * @m@ is the monad of the returned action.
  204 type ResourceCleaner k a m = k -> a -> m ()
  205 
  206 type BrokerState k a = Map k a
  207 
  208 brokerLoop ::
  209   ( MonadUnliftIO m,
  210     Ord k,
  211     IsMessageBox msgBox
  212   ) =>
  213   (forall x. m x -> m x) ->
  214   msgBox w' ->
  215   BrokerConfig k w' w a m ->
  216   BrokerState k a ->
  217   m BrokerResult
  218 brokerLoop unmask brokerBox config brokerState =
  219   ( ( unmask (receive brokerBox)
  220         >>=
  221           traverse (tryAny . onIncoming unmask config brokerState)
  222     )
  223       `onException` ( do
  224                         -- TODO logError
  225                         cleanupAllResources config brokerState
  226                     )
  227   )
  228     >>= maybe
  229       ( do
  230           -- TODO logError
  231           cleanupAllResources config brokerState
  232           return MkBrokerResult
  233       )
  234       ( \res -> do
  235           next <-
  236             either
  237               ( \_err ->
  238                   -- TODO logError
  239                   return brokerState
  240               )
  241               return
  242               res
  243           brokerLoop unmask brokerBox config next
  244       )

  246 onIncoming ::
  247   (Ord k, MonadUnliftIO m) =>
  248   (forall x. m x -> m x) ->
  249   BrokerConfig k w' w a m ->
  250   BrokerState k a ->
  251   w' ->
  252   m (BrokerState k a)
  253 onIncoming unmask config brokerState w' =
  254   case demultiplexer config w' of
  255     Initialize k mw ->
  256       onInitialize unmask k config brokerState mw
  257     Dispatch k w ->
  258       onDispatch unmask k w config brokerState
  259 
  260 onInitialize ::
  261   (Ord k, MonadUnliftIO m) =>
  262   (forall x. m x -> m x) ->
  263   k ->
  264   BrokerConfig k w' w a m ->
  265   BrokerState k a ->
  266   Maybe w ->
  267   m (BrokerState k a)
  268 onInitialize unmask k config brokerState mw =
  269   case Map.lookup k brokerState of
  270     Just _ -> do
  271       -- error "Log Error"
  272       return brokerState
  273     Nothing ->
  274       tryAny (unmask (resourceCreator config k mw))
  275         >>= either
  276           ( \_err ->
  277               -- TODO logError
  278               return brokerState
  279           )
  280           ( \res ->
  281               let brokerState1 = Map.insert k res brokerState
  282                in case mw of
  283                     Nothing ->
  284                       return brokerState1
  285                     Just w ->
  286                       onException
  287                         (onDispatch unmask k w config brokerState1)
  288                         (resourceCleaner config k res)
  289           )
  290 
  291 onDispatch ::
  292   (Ord k, MonadUnliftIO m) =>
  293   (forall x. m x -> m x) ->
  294   k ->
  295   w ->
  296   BrokerConfig k w' w a m ->
  297   BrokerState k a ->
  298   m (BrokerState k a)
  299 onDispatch unmask k w config brokerState =
  300   maybe notFound dispatch (Map.lookup k brokerState)
  301   where
  302     notFound = do
  303       -- TODO "logError"
  304       return brokerState
  305     dispatch res =
  306       tryAny (unmask (messageDispatcher config k w res))
  307         >>= either
  308           ( \_err -> do
  309               -- TODO logError
  310               cleanupResource
  311                 k
  312                 config
  313                 brokerState
  314           )
  315           ( \case
  316               KeepResource ->
  317                 return brokerState
  318               UpdateResource newRes ->
  319                 return (Map.insert k newRes brokerState)
  320               RemoveResource mNewRes ->
  321                 cleanupResource
  322                   k
  323                   config
  324                   ( maybe
  325                       brokerState
  326                       ( \newRes ->
  327                           Map.insert k newRes brokerState
  328                       )
  329                       mNewRes
  330                   )
  331           )

  333 cleanupAllResources ::
  334   (MonadUnliftIO m) =>
  335   BrokerConfig k w' w a m ->
  336   BrokerState k a ->
  337   m ()
  338 cleanupAllResources config brokerState =
  339   traverse_
  340     ( uncurry
  341         (tryResourceCleaner config)
  342     )
  343     (Map.assocs brokerState)
  344 
  345 cleanupResource ::
  346   (MonadUnliftIO m, Ord k) =>
  347   k ->
  348   BrokerConfig k w' w a m ->
  349   Map k a ->
  350   m (Map k a)
  351 cleanupResource k config brokerState = do
  352   traverse_ (tryResourceCleaner config k) (Map.lookup k brokerState)
  353   return (Map.delete k brokerState)
  354 
  355 tryResourceCleaner ::
  356   MonadUnliftIO m =>
  357   BrokerConfig k w' w a m ->
  358   k ->
  359   a ->
  360   m ()
  361 tryResourceCleaner config k res = do
  362   -- TODO logError
  363   void $ tryAny (resourceCleaner config k res)
