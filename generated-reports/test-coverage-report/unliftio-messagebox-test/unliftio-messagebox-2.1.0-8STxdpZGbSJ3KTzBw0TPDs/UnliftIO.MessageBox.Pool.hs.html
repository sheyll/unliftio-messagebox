    1 -- | Dispatch messages to processes.
    2 --
    3 -- A pool has an 'Input' for 'Multiplexed' messages,
    4 -- and dispatches incoming messges to concurrent
    5 -- processes using user defined @'MessageBox'es@.
    6 --
    7 -- The pool starts and stops the processes and
    8 -- creates the message boxes.
    9 --
   10 -- The 'PoolWorkerCallback' will process the
   11 -- 'MessageBox' until it returns, the pool
   12 -- process will then handle cleanup of the
   13 -- pool internal 'Broker' and the 'Async'
   14 -- process of the worker.
   15 module UnliftIO.MessageBox.Pool
   16   ( Pool (..),
   17     spawnPool,
   18     PoolWorkerCallback (..),
   19     removePoolWorkerMessage,
   20   )
   21 where

   53 -- | Start a 'Pool'.
   54 --
   55 -- Start a process that receives messages sent to the
   56 -- 'poolInput' and dispatches them to the 'Input' of
   57 -- __pool member__ processes. If necessary the
   58 -- pool worker processes are started.
   59 --
   60 -- Each pool worker process is started using 'async' and
   61 -- executes the 'PoolWorkerCallback'.
   62 --
   63 -- When the callback returns, the process will exit.
   64 --
   65 -- Internally the pool uses the 'async' function to wrap
   66 -- the callback.
   67 --
   68 -- When a 'Multiplixed' 'Dispatch' message is received with
   69 -- a @Nothing@ then the worker is @'cancel'led@ and the
   70 -- worker is removed from the map.
   71 --
   72 -- Such a message is automatically sent after the 'PoolWorkerCallback'
   73 -- has returned, even when an exception was thrown. See
   74 -- 'finally'.
   75 spawnPool ::
   76   forall k w poolBox workerBox m.
   77   ( IsMessageBoxArg poolBox,
   78     IsMessageBoxArg workerBox,
   79     Ord k,
   80     MonadUnliftIO m
   81   ) =>
   82   poolBox ->
   83   workerBox ->
   84   PoolWorkerCallback workerBox w k m ->
   85   m
   86     ( Either
   87         SomeException
   88         (Pool poolBox k w)
   89     )
   90 spawnPool poolBox workerBoxArg poolMemberImpl = do
   91   brInRef <- newEmptyMVar
   92   let brCfg =
   93         MkBrokerConfig
   94           id
   95           dispatchToWorker
   96           (spawnWorker workerBoxArg brInRef poolMemberImpl)
   97           removeWorker
   98   spawnBroker poolBox brCfg
   99     >>= traverse
  100       ( \(brIn, brA) -> do
  101           putMVar brInRef brIn
  102           return MkPool {poolInput = brIn, poolAsync = brA}
  103       )

  105 -- | This message will 'cancel' the worker
  106 -- with the given key.
  107 -- If the 'PoolWorkerCallback' wants to do cleanup
  108 -- it should use 'finally' or 'onException'.
  109 removePoolWorkerMessage :: k -> Multiplexed k (Maybe w)
  110 removePoolWorkerMessage !k = Dispatch k Nothing
  111 
  112 -- | The function that processes a
  113 -- 'MessageBox' of a worker for a specific /key/.
  114 newtype PoolWorkerCallback workerBox w k m = MkPoolWorkerCallback
  115   { runPoolWorkerCallback :: k -> MessageBox workerBox w -> m ()
  116   }
  117 
  118 -- | A record containing the message box 'Input' of the
  119 -- 'Broker' and the 'Async' value required to 'cancel'
  120 -- the pools broker process.
  121 data Pool poolBox k w = MkPool
  122   { -- | Message sent to this input are dispatched to workers.
  123     -- If the message is an 'Initialize' message, a new 'async'
  124     -- process will be started.
  125     -- If the message value is 'Nothing', the processes is killed.
  126     poolInput :: !(Input (MessageBox poolBox) (Multiplexed k (Maybe w))),
  127     -- | The async of the internal 'Broker'.
  128     poolAsync :: !(Async BrokerResult)
  129   }
  130 
  131 -- | Internal data structure containing a workers
  132 -- message 'Input' and 'Async' value for cancellation.
  133 data PoolWorker workerBox w = MkPoolWorker
  134   { poolWorkerIn :: !(Input (MessageBox workerBox) w),
  135     poolWorkerAsync :: !(Async ())
  136   }
  137 
  138 dispatchToWorker ::
  139   (MonadUnliftIO m, IsInput (Input (MessageBox b))) =>
  140   k ->
  141   Maybe w ->
  142   PoolWorker b w ->
  143   m (ResourceUpdate (PoolWorker b w))
  144 dispatchToWorker _k pMsg pm =
  145   case pMsg of
  146     Just w -> helper w
  147     Nothing -> return (RemoveResource Nothing)
  148   where
  149     helper msg = do
  150       ok <- deliver (poolWorkerIn pm) msg
  151       if not ok
  152         then return (RemoveResource Nothing)
  153         else return KeepResource
  154 
  155 spawnWorker ::
  156   forall k w poolBoxIn workerBox m.
  157   ( IsMessageBoxArg workerBox,
  158     MonadUnliftIO m,
  159     IsInput poolBoxIn
  160   ) =>
  161   workerBox ->
  162   MVar (poolBoxIn (Multiplexed k (Maybe w))) ->
  163   PoolWorkerCallback workerBox w k m ->
  164   k ->
  165   Maybe (Maybe w) ->
  166   m (PoolWorker workerBox w)
  167 spawnWorker workerBox brInRef pmCb this _mw = do
  168   inputRef <- newEmptyMVar
  169   a <- async (go inputRef `finally` enqueueCleanup)
  170   boxInM <- takeMVar inputRef
  171   case boxInM of
  172     Nothing -> do
  173       cancel a
  174       throwIO (stringException "failed to spawnWorker")
  175     Just boxIn ->
  176       return MkPoolWorker {poolWorkerIn = boxIn, poolWorkerAsync = a}
  177   where
  178     go inputRef = do
  179       b <-
  180         ( do
  181             b <- newMessageBox workerBox
  182             boxIn <- newInput b
  183             putMVar inputRef (Just boxIn)
  184             return b
  185           )
  186           `onException` putMVar inputRef Nothing
  187       runPoolWorkerCallback pmCb this b
  188     enqueueCleanup =
  189       tryReadMVar brInRef
  190         >>= traverse_
  191           ( \brIn ->
  192               void (deliver brIn (removePoolWorkerMessage this))
  193           )
  194 
  195 removeWorker ::
  196   (MonadUnliftIO m) =>
  197   k ->
  198   PoolWorker workerBox w ->
  199   m ()
  200 removeWorker _k =
  201   void . cancel . poolWorkerAsync
